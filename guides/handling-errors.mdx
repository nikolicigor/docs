## FileFeedError

```ts
import { FileFeedError } from '@filefeed/sdk';

try {
  await filefeed.pipelineRuns.list();
} catch (error) {
  if (error instanceof FileFeedError) {
    console.error('API Error:', error.message, error.statusCode, error.data);
  } else {
    console.error('Unexpected error:', error);
  }
}
```

## Retry with Backoff

```ts
async function withRetry<T>(fn: () => Promise<T>, max = 3, delay = 1000): Promise<T> {
  for (let i = 0; i < max; i++) {
    try { return await fn(); } 
    catch (e) { if (i === max - 1) throw e; await new Promise(r => setTimeout(r, delay * 2 ** i)); }
  }
  throw new Error('Max retries exceeded');
}
```

<Tip>
Handle 429 and 5xx responses with exponential backoff.
</Tip>
