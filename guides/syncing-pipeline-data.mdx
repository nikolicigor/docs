title: "Syncing Pipeline Data"
description: "End-to-end example to fetch processed data and acknowledge runs"

## Overview
Fetch completed runs, paginate data, store it, then acknowledge the run.

## Example (TypeScript)

```ts
import FileFeed, { PipelineRunStatus } from '@filefeed/sdk';

const filefeed = new FileFeed({ apiKey: process.env.FILEFEED_API_KEY! });

async function sync() {
  const runs = await filefeed.pipelineRuns.list({ status: PipelineRunStatus.COMPLETED, limit: 25 });
  for (const run of runs.data) {
    let offset: number | null = 0;
    let all = [] as any[];
    do {
      const page = await filefeed.pipelineRuns.getData({ pipelineRunId: run.id, limit: 1000, offset });
      all = all.concat(page.data);
      offset = page.data.length === 1000 ? (offset ?? 0) + page.data.length : null;
    } while (offset !== null);

    // store all records
    await store(all);

    await filefeed.pipelineRuns.ack({ pipelineRunId: run.id });
  }
}
```

<Tip>
Use offset-based pagination for run data (`limit` up to 1000).
</Tip>
